Multiply & divide powers (integer exponents)

\begin{aligned} \dfrac{x^n}{x^m}=x^{n-m} \\\\ \dfrac{4^{11}}{4^{-8}} &= 4^{11-(-8)} \\\\ &= 4^{19} \end{aligned}
\begin{aligned} \dfrac{z^{7}}{z^{-14}}&=z^{7-(-14)} \\\\ &= z^{7+14} \\\\ &= z^{21} \end{aligned}
\begin{aligned} x^n\cdot x^m=x^{n+m} \\\\ y^{-4}\cdot y^{3}&=y^{-4+3} \\\\ &= y^{-1} \end{aligned}


Powers of products & quotients (integer exponents)

\begin{aligned} \left(6^{-4}\cdot 8^{-7}\right)^{-9}&=\left(6^{-4}\right)^{-9}\cdot\left(8^{-7}\right)^{-9} \end{aligned} \begin{aligned} \phantom{\left(6^{-4}\cdot 8^{-7}\right)^{-9}}&=6^{(-4)(-9)}\cdot 8^{(-7)(-9)} \\\\ &=6^{36}\cdot 8^{63} \end{aligned}
\begin{aligned} \left(\dfrac{8^{-5}}{2^{-2}}\right)^{-4}&=\dfrac{\left(8^{-5}\right)^{-4}}{\left(2^{-2}\right)^{-4}} \end{aligned} \begin{aligned} \phantom{\left(\dfrac{8^{-5}}{2^{-2}}\right)^{-4}}&=\dfrac{8^{(-5)(-4)}}{2^{(-2)(-4)}} \\\\ &=\dfrac{8^{20}}{2^{8}} \end{aligned}
\begin{aligned} {\left( 5^3 \cdot 5^2 \right)^{4}} &\stackrel{?}{=} 25^{20} \\\\ {\left( 5^{3+2} \right)^{4}} &\stackrel{?}{=} 25^{20} \\\\ {\left( 5^{5}\right)^{4}} &\stackrel{?}{=} 25^{20} \\\\ {5^{5 \: \cdot \: 4}} &\stackrel{?}{=} 25^{20} \\\\ {5^{20}} &\neq 25^{20} \end{aligned}
\begin{aligned} {\left( 5^3 \cdot 5^2 \right)^{4}} &\stackrel{?}{=} \left( 25^{5} \right)^{4} \\\\ {\left( 5^{3+2} \right)^{4}} &\stackrel{?}{=} \left( 25^{5} \right)^{4} \\\\ {\left( 5^{5}\right)^{4}} &\neq \left( 25^{5} \right)^{4} \end{aligned}
\begin{aligned} {4^{-2} \cdot 7^{-2}} &\stackrel{?}{=} (4 \cdot 7)^{-4} \\\\ {(4 \cdot 7)^{-2}} &\neq (4 \cdot 7)^{-4} \end{aligned}
\begin{aligned} {4^{-2} \cdot 7^{-2}} &\stackrel{?}{=} \dfrac{1}{28^2} \\\\ {(4 \cdot 7)^{-2}} &\stackrel{?}{=} \dfrac{1}{28^2} \\\\ {28^{-2}} &\stackrel{?}{=} \dfrac{1}{28^2} \\\\ {\dfrac{1}{28^2}} &= \dfrac{1}{28^2} \end{aligned}
