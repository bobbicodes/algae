Evaluating radical expressions
\begin{aligned} \dfrac{\sqrt[3]250}{\sqrt[3]2} &=\sqrt[3]{\dfrac{250}{2}} \\\\ &=\sqrt[3]125 \\\\ &=5 \end{aligned}
\begin{aligned} \dfrac{256^{\frac{4}{7}}}{2^{\frac{4}{7}}} &=\left(\dfrac{256}{2}\right)^{\frac{4}{7}} \\\\ &=128^{\frac47} \\\\ &=\left(\sqrt[7]{128}\right)^4 \\\\ &=2^4 \\\\ &=16 \end{aligned}
\begin{aligned} \dfrac{\sqrt[3]{2}}{\sqrt[3]{-128}}&=\sqrt[3]{\dfrac{2}{-128}} \\\\ &=\sqrt[3]{-\dfrac{1}{64}} \\\\ &=-\dfrac{1}{\sqrt[3]{64}} \\\\ &=-\dfrac14 \end{aligned}
\begin{aligned} \dfrac{2^{-\frac43}}{54^{-\frac43}} &=\left(\dfrac{2}{54}\right)^{-\frac43} \\\\ &=\left(\dfrac{1}{27}\right)^{-\frac43} \\\\ &=27^{\frac43} \\\\ &=\left(\sqrt[3]{27}\right)^4 \\\\ &=3^4 \\\\ &=81 \end{aligned}
\begin{aligned} \left(\dfrac{1}{81}\right)^\frac12\cdot\left(\dfrac{1}{81}\right)^{-\frac34} &=\left(\dfrac{1}{81}\right)^{\frac12+\left(-\frac34\right)} \\\\ &=\left(\dfrac{1}{81}\right)^{\frac24-\frac34} \\\\ &=\left(\dfrac{1}{81}\right)^{-\frac14} \\\\ &=81^{\frac14} \\\\ &=\sqrt[4]81 \\\\ &=3 \end{aligned}
\begin{aligned} 324^{\frac14}\cdot\sqrt[4]{\dfrac14} &=\sqrt[4]324\cdot\sqrt[4]{\dfrac14} \\\\ &=\sqrt[4]{324\cdot\dfrac14} \\\\ &=\sqrt[4]81 \\\\ &=3 \end{aligned}
\begin{aligned} \sqrt[3]-54\cdot\sqrt[3]{\dfrac12} &=\sqrt[3]{-54\cdot\dfrac12} \\\\ &=\sqrt[3]-27 \\\\ &=-3 \end{aligned}
\begin{aligned} \dfrac{\sqrt{98}}{2^\frac12} &=\dfrac{\sqrt{98}}{\sqrt2} \\\\ &=\sqrt{\frac{98}{2}} \\\\ &=\sqrt49 \\\\ &=7 \end{aligned}
\begin{aligned} \left(-7\right)^{\frac53}\cdot\left(\dfrac{1}{56}\right)^{\frac53} &=\left(-\dfrac{7}{56}\right)^\frac53 \\\\ &=\left(-\dfrac18\right)^\frac53 \\\\ &=\dfrac{1}{\left(-8\right)^\frac53} \\\\ &=\dfrac{1}{\left(\sqrt[3]{-8}\right)^5} \\\\ &=\dfrac{1}{(-2)^5} \\\\ &=-\dfrac{1}{32} \end{aligned}
\begin{aligned} 243^{\frac{7}{10}}\cdot 243^{\frac{1}{10}} &=243^{\frac{7}{10}+\frac{1}{10}} \\\\ &=243^{\frac{8}{10}} \\\\ &=243^{\frac45} \\\\ &=\left(\sqrt[5]{243}\right)^4 \\\\ &=3^4 \\\\ &=81 \end{aligned}
\begin{aligned} \dfrac{125^{-\frac{11}{12}}}{125^{-\frac14}} &=125^{-\frac{11}{12}-\left(-\frac14\right)} \\\\ &=125^{-\frac{11}{12}+\frac{3}{12}} \\\\ &=125^{-\frac{8}{12}} \\\\ &=125^{-\frac23} \\\\ &=\dfrac{1}{125^{\frac23}} \\\\ &=\dfrac{1}{\left(\sqrt[3]{125}\right)^2} \\\\ &=\dfrac{1}{5^2} \\\\ &=\dfrac{1}{25} \end{aligned}
\begin{aligned} \dfrac{3^{\frac15}}{\sqrt[5]{-96}} &=\dfrac{\sqrt[5]{3}}{\sqrt[5]{-96}} \\\\ &=\sqrt[5]{\dfrac{3}{-96}} \\\\ &=\sqrt[5]{-\dfrac{1}{32}} \\\\ &=-\dfrac{1}{\sqrt[5]{32}} \\\\ &=-\dfrac12 \end{aligned}
\begin{aligned} \dfrac{\sqrt[4]{2}}{\sqrt[4]{162}} &=\sqrt[4]{\dfrac{2}{162}} \\\\ &=\sqrt[4]{\dfrac{1}{81}} \\\\ &=\dfrac{1}{\sqrt[4]{81}} \\\\ &=\dfrac13 \end{aligned}
\begin{aligned} \sqrt[3]{-3} \cdot \left({\dfrac{1}{375}}\right)^{\frac13} &=\sqrt[3]{-3} \cdot \sqrt[3]{\dfrac{1}{375}} \\\\ &=\sqrt[3]{-3 \cdot {\dfrac{1}{375}}} \\\\ &=\sqrt[3]{-{\dfrac{1}{125}}} \\\\ &=-{\dfrac{1}{\sqrt[3]{125}}} \\\\ &=-\dfrac15 \end{aligned}
\begin{aligned} \dfrac{\sqrt[3]{-5}}{40^{\frac13}} &=\dfrac{\sqrt[3]{-5}}{\sqrt[3]{40}} \\\\ &=\sqrt[3]{\dfrac{-5}{40}} \\\\ &=\sqrt[3]{-\dfrac18} \\\\ &=-\dfrac{1}{\sqrt[3]{8}} \\\\ &=-\dfrac12 \end{aligned}
\begin{aligned} \left(\dfrac13\right)^{-\frac25}\cdot 96^{^{-\frac25}} &=\left({\dfrac{96}{3}}\right)^{-\frac25} \\\\ &=32^{^{-\frac25}} \\\\ &=\dfrac{1}{32^\frac25} \\\\ &=\dfrac{1}{\left(\sqrt[5]{32}\right)^2} \\\\ &=\dfrac{1}{2^2} \\\\ &=\dfrac14 \end{aligned}
\begin{aligned} \sqrt[5]{-16}\cdot\sqrt[5]{2} &=\sqrt[5]{-16\cdot 2} \\\\ &=\sqrt[5]{-32} \\\\ &=-2 \end{aligned}
\begin{aligned} \sqrt[4]{\dfrac13}\cdot\sqrt[4]{48} &=\sqrt[4]{\dfrac13\cdot 48} \\\\ &=\sqrt[4]{16} \\\\ &=2 \end{aligned}
\begin{aligned} \dfrac{64^{^{\frac{7}{10}}}}{64^{^{\frac15}}} &=64^{\frac{7}{10}-\frac15} \\\\ &=64^{\frac{7}{10}-\frac{2}{10}} \\\\ &=64^{\frac{5}{10}} \\\\ &=64^{\frac12} \\\\ &=\sqrt{64} \\\\ &=8 \end{aligned}
